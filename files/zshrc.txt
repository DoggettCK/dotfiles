export ZSH=~/.oh-my-zsh
export ZSH_THEME="gentoo"
export EDITOR='vim'
export LANG=en_US.UTF-8
export DEFAULT_USER=`whoami`

if [[ $(uname) == 'Linux' ]]; then
    export TERM=xterm-256color
fi

plugins=(bundler jump encode64 httpie rails urltools web-search)

# requires 'jump' plugin above
alias j='jump'

# User configuration
source $ZSH/oh-my-zsh.sh

if which git-number > /dev/null; then
	alias gn='git number --column'
	alias gna='git number add'
fi

if which tmate > /dev/null; then
  alias tmlink="tmate show-messages | grep 'ssh session: ssh -p' | tail -1 | sed -e 's/^.*session: //' | pbcopy"
fi

if [[ -e /usr/local/bin/emacs ]]; then
  alias emacs='/usr/local/bin/emacs -nw'
fi

if [[ -e $HOME/Applications/exercism ]]; then
  alias exercism='$HOME/Applications/exercism'
fi

if which xml2 > /dev/null; then
  function podcast_rss_to_curl() {
    curl -s $1 |
    xml2 |
    grep 'enclosure/@url' |
    cut -d'=' -f2 |
    sed 's/\(.*\)/curl -LO \1/'
  }

  alias pod_nerdist="podcast_rss_to_curl 'http://nerdist.libsyn.com/rss'"
  alias pod_mbmbam="podcast_rss_to_curl 'https://feeds.feedburner.com/mbmbam'"
  alias pod_adventurezone="podcast_rss_to_curl 'http://adventurezone.libsyn.com/rss'"
fi

source $HOME/.dotfiles/zsh/config.sh
source $HOME/.dotfiles/zsh/aliases.sh
source $HOME/.dotfiles/zsh/functions.sh
source $HOME/.dotfiles/shared/aliases.sh
source $HOME/.dotfiles/shared/functions.sh
