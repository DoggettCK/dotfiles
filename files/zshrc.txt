export ZSH=~/.oh-my-zsh
export ZSH_THEME="gentoo"
export EDITOR='vim'
export LANG=en_US.UTF-8
export DEFAULT_USER=`whoami`
export FZF_DEFAULT_COMMAND='rg --files'

if [[ $(uname) == 'Linux' ]]; then
    export TERM=xterm-256color
fi

plugins=(bundler jump encode64 httpie rails urltools web-search)

source $ZSH/oh-my-zsh.sh

# Common user configuration
source $HOME/.dotfiles/zsh/config.sh
source $HOME/.dotfiles/zsh/functions.sh
source $HOME/.dotfiles/shared/functions.sh
source $HOME/.dotfiles/zsh/aliases.sh
source $HOME/.dotfiles/shared/aliases.sh

# My aliases

if which jump > /dev/null; then
  alias j='jump'
fi

if [[ -e $HOME/git-number ]]; then
  if ! which git-number > /dev/null; then
    export PATH=$PATH:$HOME/git-number
  fi
fi

if which git-number > /dev/null; then
	alias gn='git number --column'
	alias ga='git number add'
fi

if which tmate > /dev/null; then
  function tmlink() {
    tmate show-messages |
      grep 'ssh session: ssh -p' |
      tail -1 |
      sed -e 's/^.*session: //' |
      pbcopy
  }
fi

if [[ -e /usr/local/bin/emacs ]]; then
  alias emacs='/usr/local/bin/emacs -nw'
fi

if [[ -e $HOME/Applications/exercism ]]; then
  alias exercism='$HOME/Applications/exercism'
fi

if which xml2 > /dev/null; then
  function podcast() {
    curl -s $1 |
      xml2 |
      grep 'enclosure/@url' |
      cut -d'=' -f2 |
      sed 's/\(.*\)/curl -LO \1/' |
      sed 's/\?dest-id$//g'
  }

  alias mbmbam="podcast 'https://feeds.feedburner.com/mbmbam'"
  alias adventurezone="podcast 'http://adventurezone.libsyn.com/rss'"
  alias beastcast="podcast 'https://www.giantbomb.com/podcast-xml/beastcast/'"
  alias giantbomb="podcast 'https://www.giantbomb.com/feeds/podcast/'"
  alias occultae="podcast 'http://ovpod.libsyn.com/rss'"
fi

[ -f ~/.fzf.zsh ] && source ~/.fzf.zsh
